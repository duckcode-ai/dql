Notebook cell: retention_driver_preview.sql
Review-required Inserted by chat 
with retention_base as (
  select
    account_segment,
    product_family,
    contraction_reason,
    sum(expansion_arr) as expansion_arr,
    sum(contraction_arr) as contraction_arr,
    sum(churn_arr) as churn_arr
  from marts.customer_retention_monthly
  where month between date '2026-01-01' and date '2026-03-31'
  group by 1, 2, 3
)
select
  account_segment,
  product_family,
  contraction_reason,
  expansion_arr,
  contraction_arr,
  churn_arr,
  expansion_arr - contraction_arr - churn_arr as net_arr_impact
from retention_base
order by abs(net_arr_impact) desc
limit 50;
